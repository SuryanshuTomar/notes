Q16: What happens if you declare a variable with the same name in different scopes?

You can declare variables with the same name in different (nested) scopes. Each variable is only accessible within its own scope.

let outer = 'outer';
{
    let outer = 'inner';
    {
        let outer = 'innermost';
        console.log(outer); // 'innermost'
    }
    console.log(outer); // 'inner'
}
console.log(outer); // 'outer'
