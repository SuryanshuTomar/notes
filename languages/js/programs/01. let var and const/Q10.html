Q10: What is the scope of var, let, and const?

var is function scoped, so it is accessible anywhere within the function. let and const are block scoped, so they are only accessible within the block they are defined.

function scopeTest() {
    if (true) {
        var x = 1;
        let y = 2;
        const z = 3;
    }
    console.log(x); // 1 (function scoped)
    // console.log(y); // ReferenceError (block scoped)
    // console.log(z); // ReferenceError (block scoped)
}
scopeTest();
